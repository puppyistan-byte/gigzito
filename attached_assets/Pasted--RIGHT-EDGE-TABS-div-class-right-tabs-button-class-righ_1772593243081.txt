<script>
  const openBtn = document.getElementById("openMarketers");
  const closeBtn = document.getElementById("closeMarketers");
  const overlay = document.getElementById("mkOverlay");
  const drawer = document.getElementById("mkDrawer");
  const carousel = document.getElementById("mkCarousel");
  const dotsWrap = document.getElementById("mkDots");

  // build dots dynamically based on number of cards
  function buildDots(){
    const cards = carousel.children.length;
    dotsWrap.innerHTML = "";
    for(let i=0;i<cards;i++){
      const d=document.createElement("span");
      d.className="mk-dot" + (i===0?" active":"");
      d.addEventListener("click", ()=> carousel.scrollTo({left: i*carousel.clientWidth, behavior:"smooth"}));
      dotsWrap.appendChild(d);
    }
  }

  function setOpen(isOpen){
    overlay.classList.toggle("open", isOpen);
    drawer.classList.toggle("open", isOpen);
    drawer.setAttribute("aria-hidden", String(!isOpen));
  }

  openBtn.addEventListener("click", ()=> {
    buildDots();
    setOpen(true);
  });

  [closeBtn, overlay].forEach(el => el.addEventListener("click", ()=> setOpen(false)));

  carousel.addEventListener("scroll", () => {
    const idx = Math.round(carousel.scrollLeft / carousel.clientWidth);
    const dots = [...dotsWrap.querySelectorAll(".mk-dot")];
    dots.forEach((d,i)=> d.classList.toggle("active", i===idx));
  });

  document.addEventListener("keydown", (e)=> {
    if(e.key==="Escape") setOpen(false);
  });
</script>
