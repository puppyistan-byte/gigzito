GZMusic Mobile API — all endpoints public, user context optional (session cookie enriches liked / myRating)

GET	/api/gz-music/genres	—	string[]
GET	/api/gz-music/tracks/:id	—	single track or 404
POST	/api/gz-music/tracks/:id/play	—	increments play count
POST	/api/gz-music/tracks/:id/like	auth required	{ liked, likeCount }
POST	/api/gz-music/tracks/:id/rate	body: { stars }, auth required	{ avgRating, ratingCount, myRating }
GET	/api/gz-music/tracks/:id/comments	—	comment[] with displayName/avatar
POST	/api/gz-music/tracks/:id/comments	body: { content }, auth required	new comment
