<script>
  const dock = document.getElementById("edgeDock");
  const closeBtn = document.getElementById("edgeClose");
  const carousel = document.getElementById("edgeCarousel");
  const dots = Array.from(document.querySelectorAll(".edgeDots .dot"));

  // open on hover (desktop)
  dock.addEventListener("mouseenter", () => dock.classList.add("open"));
  dock.addEventListener("mouseleave", () => dock.classList.remove("open"));

  // close button (works for touch devices too)
  closeBtn.addEventListener("click", (e) => {
    e.stopPropagation();
    dock.classList.remove("open");
  });

  // dot syncing
  carousel.addEventListener("scroll", () => {
    const idx = Math.round(carousel.scrollLeft / carousel.clientWidth);
    dots.forEach((d, i) => d.classList.toggle("active", i === idx));
  });
</script>
